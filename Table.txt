CREATE TABLE KONTRAK
(
    KONTRAK_NO VARCHAR(20) PRIMARY KEY,
    CLIENT_NAME VARCHAR(100),
    OTR DECIMAL(18,0),
    DP DECIMAL(18,0),
    POKOK_UTANG DECIMAL(18,0),
    TENOR INT,
    BUNGA DECIMAL(5,2),
    TOTAL_HUTANG DECIMAL(18,0),
    ANGSURAN_PER_BULAN DECIMAL(18,0),
    START_DATE DATE
)

CREATE TABLE JADWAL_ANGSURAN
(
    ID INT IDENTITY(1,1) PRIMARY KEY,
    KONTRAK_NO VARCHAR(20),
    ANGSURAN_KE INT,
    ANGSURAN_PER_BULAN DECIMAL(18,0),
    TANGGAL_JATUH_TEMPO DATE,

    FOREIGN KEY (KONTRAK_NO)
    REFERENCES KONTRAK(KONTRAK_NO)
)
